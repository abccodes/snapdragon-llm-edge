```python
# Create a function that checks if a given string is a palindrome
def check_palindrome(s):
    # Normalize the input string by removing spaces and converting to lowercase
    normalized_string = s.replace(" ", "").lower()
    
    # Check if the normalized string is equal to its reverse
    return normalized_string == normalized_string[::-1]

# Test the function with different inputs
print(check_palindrome('A man a plan a canal Panama'))  # True
print(check_palindrome('Hello World'))                # False
```
