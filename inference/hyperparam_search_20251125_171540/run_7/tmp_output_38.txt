import spacy
def answer_fact_question(sentence):
    # Tokenize sentence and identify named entities
    nlp = spacy.load('en_core_web_sm')
    tokens = nlp.tokenization()
    
    # Identify country entity from tokenized text
    for i in range(len(tokens)):
        doc = tokens[i]
        
        if 'country' in doc:
            return doc['name']

# Test the function with a sample sentence
