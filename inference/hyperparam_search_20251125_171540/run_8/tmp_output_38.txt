```python
import requests

def get_country_of_birth(country_name):
    url = "https://api.geonames.org/getCountryByCode?name=" + country_name
    response = requests.get(url)
    data = response.json()
    return data['country']

# Example usage:
print(get_country_of_birth("Barack Obama"))
```
