# Define a function that takes a quote as input and returns the author.
def get_author_of_quote(quote):
    # List of famous quotes with authors.
    known_quotes = [
        ('You cannot find peace by avoiding life', 'Mahatma Gandhi'),
        ('The only way to do great work is to love what you do', 'Steve Jobs'),
        ('A person who never made a mistake never tried anything new', 'Albert Einstein')
    ]
    
    # Check each quote in the list to see if it matches.
    for q, author in known_quotes:
        if q == quote:
            return author
