import requests
from bs4 import BeautifulSoup

response = requests.get("https://www.usda.gov/water/national-plant-production")
soup = BeautifulSoup(response.text, "html.parser")

peach_states = soup.find_all("table", class_="production-data-table")

data = peach_states.find("tr", class_="data-row production-data-table")
rows = data.find_all("td", class_="data-cell production-data-table")

max_production = 0
state = ""

for row in rows:
    production = row.text
    state = row.next().text

if production > max_production:
        max_production = production
        state = row.next().text

print("The state that produces the most peaches is:", state)
